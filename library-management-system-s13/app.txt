function StatusBadge({ status }: { status: string }) {
  const map: Record<string, string> = {
    active: "bg-green-50 text-green-700 border-green-200 dark:bg-green-900/30 dark:text-green-400 dark:border-green-800",
    overdue: "bg-red-50 text-red-600 border-red-200 dark:bg-red-900/30 dark:text-red-400 dark:border-red-800",
    returned: "bg-slate-100 text-slate-500 border-slate-200 dark:bg-slate-700 dark:text-slate-400 dark:border-slate-600",
    pending: "bg-amber-50 text-amber-700 border-amber-200 dark:bg-amber-900/30 dark:text-amber-400 dark:border-amber-800",
    approved: "bg-blue-50 text-blue-700 border-blue-200 dark:bg-blue-900/30 dark:text-blue-400 dark:border-blue-800",
    expired: "bg-slate-100 text-slate-500 border-slate-200 dark:bg-slate-700 dark:text-slate-400 dark:border-slate-600",
    suspended: "bg-red-50 text-red-600 border-red-200 dark:bg-red-900/30 dark:text-red-400 dark:border-red-800",
  };
  const labels: Record<string, string> = { active: "Active", overdue: "Overdue", returned: "Returned", pending: "Pending", approved: "Approved", expired: "Expired", suspended: "Suspended" };
  return (
    <span className={`inline-flex items-center px-2 py-0.5 rounded-full text-xs font-semibold border ${map[status] ?? map.active}`}>
      {labels[status] ?? status}
    </span>
  );
}

function BookCard({ book, onView, onBorrow, onReserve }: {
  book: Book;
  onView?: () => void; onBorrow?: () => void; onReserve?: () => void;
}) {
  const stars = Math.round(book.rating);
  return (
    <article className="bg-card rounded-2xl overflow-hidden border border-border shadow-sm hover:shadow-lg transition-all hover:-translate-y-1 flex flex-col group">
      <div className="relative h-44 bg-muted overflow-hidden flex-shrink-0">
        <img src={book.cover} alt={`Cover of ${book.title}`} className="w-full h-full object-cover group-hover:scale-105 transition-transform duration-500" loading="lazy" />
        <div className="absolute top-2 right-2">
          {book.available > 0
            ? <span className="px-2 py-0.5 bg-green-500 text-white text-xs font-bold rounded-full shadow-sm">Available</span>
            : <span className="px-2 py-0.5 bg-red-500 text-white text-xs font-bold rounded-full shadow-sm">Out of Stock</span>}
        </div>
        <div className="absolute top-2 left-2 flex gap-0.5">
          {Array.from({ length: 5 }).map((_, i) => (
            <Star key={i} className={`w-3 h-3 ${i < stars ? "fill-amber-400 text-amber-400" : "text-white/40"}`} aria-hidden="true" />
          ))}
        </div>
      </div>
      <div className="p-4 flex flex-col flex-1">

        <p className="text-xs text-muted-foreground mb-2">{book.author}</p>

          </span>
          <span className="text-xs text-muted-foreground bg-muted px-2 py-0.5 rounded-full">{book.shelf}</span>

function Modal({ open, onClose, title, children }: {
  open: boolean; onClose: () => void; title: string; children: React.ReactNode;
}) {
  if (!open) return null;
  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center p-4" role="dialog" aria-modal="true" aria-labelledby="modal-title">
      <div className="absolute inset-0 bg-black/50 backdrop-blur-sm" onClick={onClose} aria-hidden="true" />
      <div className="relative bg-card rounded-2xl shadow-2xl w-full max-w-lg border border-border max-h-[90vh] overflow-y-auto">
        <div className="flex items-center justify-between px-6 py-4 border-b border-border sticky top-0 bg-card z-10">
          <h2 id="modal-title" className="font-bold text-foreground" style={{ fontFamily: "Poppins, sans-serif" }}>{title}</h2>
          <button onClick={onClose} className="p-1.5 rounded-lg text-muted-foreground hover:bg-muted transition-colors" aria-label="Close dialog"><X className="w-4 h-4" /></button>
        </div>
        <div className="p-6">{children}</div>
      </div>
    </div>
  );
}

}) {

      </div>
    </div>
  );
}

            </div>

        </div>
      </div>
    </div>
  );
}

// ─── Screen: Browse Books ─────────────────────────────────────────────────────

function BrowseBooks({ onViewBook }: { onViewBook: (b: Book) => void }) {
  const [q, setQ] = useState("");
  const [cat, setCat] = useState("All");
  const [avail, setAvail] = useState(false);
  const cats = ["All", "Computer Science", "Software Engineering", "Literature", "Economics", "Chemistry", "Physics", "Self Help"];
  const filtered = BOOKS.filter((b) => {
    const mq = b.title.toLowerCase().includes(q.toLowerCase()) || b.author.toLowerCase().includes(q.toLowerCase()) || b.isbn.includes(q);
    return mq && (cat === "All" || b.category === cat) && (!avail || b.available > 0);
  });

  return (
    <div className="space-y-5">
      <div className="bg-card rounded-2xl p-5 border border-border shadow-sm">
        <div className="flex flex-col sm:flex-row gap-3">
          <div className="relative flex-1">
            <Search className="w-4 h-4 absolute left-3 top-1/2 -translate-y-1/2 text-muted-foreground/60 pointer-events-none" aria-hidden="true" />
            <input value={q} onChange={(e) => setQ(e.target.value)} placeholder="Search by title, author, or ISBN…" aria-label="Search books" className="w-full pl-9 pr-4 py-2.5 text-sm border border-border rounded-xl bg-input-background text-foreground placeholder:text-muted-foreground/50 focus:outline-none focus:ring-2 focus:ring-ring/40 focus:border-primary transition" />
          </div>
          <label className="flex items-center gap-2 text-sm text-foreground cursor-pointer px-4 py-2 border border-border rounded-xl hover:bg-muted transition-colors select-none">
            <input type="checkbox" checked={avail} onChange={(e) => setAvail(e.target.checked)} className="accent-primary" />
            Available only
          </label>
        </div>
        <div className="flex gap-2 mt-3 flex-wrap" role="group" aria-label="Filter by category">
          {cats.map((c) => (
            <button key={c} onClick={() => setCat(c)} aria-pressed={cat === c} className={`px-3 py-1 rounded-full text-xs font-semibold transition-colors ${cat === c ? "bg-primary text-primary-foreground" : "bg-muted text-muted-foreground hover:bg-muted/70"}`}>
              {c}
            </button>
          ))}
        </div>
      </div>

      <p className="text-sm text-muted-foreground"><span className="font-bold text-foreground">{filtered.length}</span> books found</p>

      {filtered.length > 0 ? (
        <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-5">
          {filtered.map((book) => (
            <BookCard key={book.id} book={book} onView={() => onViewBook(book)} onBorrow={() => toast.success(`"${book.title}" added to your loans.`)} onReserve={() => toast.success(`"${book.title}" reserved.`)} />
          ))}
        </div>
      ) : (
        <EmptyState icon={BookOpen} title="No books found" desc="Try adjusting your search or filters" />
      )}
    </div>
  );
}

    const m = MEMBERS.find((m) => m.name.toLowerCase().includes(memberQ.toLowerCase()) || m.id.toLowerCase() === memberQ.toLowerCase());
    setFoundMember(m ?? null);
    if (!m) toast.error("No member found.");
  };
  const searchBook = () => {
    const b = BOOKS.find((b) => b.title.toLowerCase().includes(bookQ.toLowerCase()) || b.isbn.includes(bookQ));
    setFoundBook(b ?? null);
    if (!b) toast.error("No book found.");
  };
  const handleIssue = () => {
    if (!foundMember || !foundBook) { toast.error("Select a member and book first."); return; }
    if (foundBook.available === 0) { toast.error("No copies available."); return; }
    toast.success(`"${foundBook.title}" issued to ${foundMember.name}.`);
    setFoundMember(null); setFoundBook(null); setMemberQ(""); setBookQ("");
  };
  const handleReturn = () => {
    if (!foundBook) { toast.error("Select the book to return."); return; }
    toast.success(`"${foundBook.title}" returned.`);
    setFoundBook(null); setBookQ("");
  };

  return (
    <div className="space-y-5">
      <div className="flex gap-1 p-1 bg-card rounded-xl border border-border shadow-sm w-fit" role="tablist">
        {(["issue", "return"] as const).map((t) => (
          <button key={t} role="tab" aria-selected={tab === t} onClick={() => setTab(t)} className={`px-5 py-2 text-sm font-bold rounded-lg capitalize transition-all ${tab === t ? "bg-primary text-primary-foreground shadow-sm" : "text-muted-foreground hover:text-foreground"}`}>
            {t === "issue" ? "Issue Book" : "Return Book"}
          </button>
        ))}
      </div>

      <div className="grid grid-cols-1 lg:grid-cols-3 gap-5">
        {/* QR Scanner */}
        <div className="bg-card rounded-2xl p-5 border border-border shadow-sm">
          <h2 className="text-sm font-bold text-muted-foreground mb-4 uppercase tracking-wide flex items-center gap-2">
            <QrCode className="w-4 h-4" aria-hidden="true" /> QR / Barcode Scanner
          </h2>
          <div className="relative bg-muted rounded-2xl h-40 flex items-center justify-center border-2 border-dashed border-border overflow-hidden" aria-label="QR scanner area" role="img">
            {scanning ? (
              <div className="flex flex-col items-center gap-2">
                <div className="w-12 h-12 border-4 border-primary border-t-transparent rounded-full animate-spin" role="status" aria-label="Scanning" />
                <p className="text-xs text-muted-foreground">Scanning…</p>
              </div>
            ) : (
              <div className="flex flex-col items-center gap-2 text-muted-foreground">
                <QrCode className="w-12 h-12 opacity-25" aria-hidden="true" />
                <p className="text-xs">Click to scan</p>
              </div>
            )}
          </div>
          <button onClick={() => { setScanning(true); setTimeout(() => { setScanning(false); toast.success("QR scanned! Book identified."); }, 2000); }} className="w-full mt-3 py-2.5 bg-primary/10 text-primary font-bold text-sm rounded-xl hover:bg-primary/20 transition-colors" aria-label="Start QR scan">
            {scanning ? "Scanning…" : "Start QR Scan"}
          </button>
        </div>

        {/* Member Search */}
        <div className="bg-card rounded-2xl p-5 border border-border shadow-sm">
          <h2 className="text-sm font-bold text-muted-foreground mb-4 uppercase tracking-wide flex items-center gap-2">
            <User className="w-4 h-4" aria-hidden="true" /> Search Member
          </h2>
          <div className="flex gap-2 mb-3">
            <input value={memberQ} onChange={(e) => setMemberQ(e.target.value)} onKeyDown={(e) => e.key === "Enter" && searchMember()} placeholder="Member ID or name…" aria-label="Search member" className="flex-1 min-w-0 px-3 py-2 text-sm border border-border rounded-xl bg-input-background text-foreground placeholder:text-muted-foreground/50 focus:outline-none focus:ring-2 focus:ring-ring/40 focus:border-primary transition" />
            <button onClick={searchMember} className="px-3 py-2 bg-primary text-primary-foreground text-sm font-bold rounded-xl hover:bg-primary/90 transition-colors">Find</button>
          </div>
          {foundMember ? (
            <div className="p-3 bg-muted rounded-xl border border-border">
              <div className="flex items-center gap-3">
                <div className="w-9 h-9 rounded-full bg-primary flex items-center justify-center text-white text-xs font-bold flex-shrink-0" aria-hidden="true">
                  {foundMember.name.split(" ").map((n) => n[0]).join("").slice(0, 2)}
                </div>
                <div>
                  <p className="text-sm font-bold text-foreground">{foundMember.name}</p>
                  <p className="text-xs text-muted-foreground">{foundMember.id} · {foundMember.dept}</p>
                </div>
              </div>
              <div className="flex items-center gap-4 mt-2 pt-2 border-t border-border text-xs text-muted-foreground">
                <span>{foundMember.borrowed} borrowed</span>
                <StatusBadge status={foundMember.status} />
              </div>
            </div>
          ) : (
            <div className="p-3 bg-muted rounded-xl border border-dashed border-border text-center text-xs text-muted-foreground">Enter ID or name and click Find</div>
          )}
        </div>

        {/* Book Search */}
        <div className="bg-card rounded-2xl p-5 border border-border shadow-sm">
          <h2 className="text-sm font-bold text-muted-foreground mb-4 uppercase tracking-wide flex items-center gap-2">
            <BookOpen className="w-4 h-4" aria-hidden="true" /> Search Book
          </h2>
          <div className="flex gap-2 mb-3">
            <input value={bookQ} onChange={(e) => setBookQ(e.target.value)} onKeyDown={(e) => e.key === "Enter" && searchBook()} placeholder="Title or ISBN…" aria-label="Search book" className="flex-1 min-w-0 px-3 py-2 text-sm border border-border rounded-xl bg-input-background text-foreground placeholder:text-muted-foreground/50 focus:outline-none focus:ring-2 focus:ring-ring/40 focus:border-primary transition" />
            <button onClick={searchBook} className="px-3 py-2 bg-primary text-primary-foreground text-sm font-bold rounded-xl hover:bg-primary/90 transition-colors">Find</button>
          </div>
          {foundBook ? (
            <div className="p-3 bg-muted rounded-xl border border-border">
              <div className="flex items-center gap-3">
                <div className="w-9 h-12 rounded-lg overflow-hidden bg-card flex-shrink-0">
                  <img src={foundBook.cover} alt={`Cover of ${foundBook.title}`} className="w-full h-full object-cover" loading="lazy" />
                </div>
                <div className="flex-1 min-w-0">
                  <p className="text-sm font-bold text-foreground truncate">{foundBook.title}</p>
                  <p className="text-xs text-muted-foreground">{foundBook.author}</p>
                  <p className="text-xs mt-1">
                    <span className={`font-semibold ${foundBook.available > 0 ? "text-green-600 dark:text-green-400" : "text-red-500"}`}>{foundBook.available}</span>
                    <span className="text-muted-foreground">/{foundBook.total} available</span>
                  </p>
                </div>
              </div>
            </div>
          ) : (
            <div className="p-3 bg-muted rounded-xl border border-dashed border-border text-center text-xs text-muted-foreground">Enter title or ISBN and click Find</div>
          )}
        </div>
      </div>

      {/* Action */}
      <div className="bg-card rounded-2xl p-5 border border-border shadow-sm flex flex-col sm:flex-row items-center gap-4">
        <p className="flex-1 text-sm text-muted-foreground">
          {tab === "issue"
            ? foundMember && foundBook ? `Ready to issue "${foundBook.title}" to ${foundMember.name}` : "Select a member and book above"
            : foundBook ? `Ready to return "${foundBook.title}"` : "Select a book to return"}
        </p>
        <div className="flex gap-3">
          {tab === "issue" && (
            <button onClick={handleIssue} disabled={!foundMember || !foundBook} className="px-6 py-2.5 bg-green-600 text-white font-bold rounded-xl hover:bg-green-700 transition-all disabled:opacity-40 disabled:cursor-not-allowed shadow-md">Issue Book</button>
          )}
          {tab === "return" && (
            <button onClick={handleReturn} disabled={!foundBook} className="px-6 py-2.5 bg-amber-500 text-white font-bold rounded-xl hover:bg-amber-600 transition-all disabled:opacity-40 disabled:cursor-not-allowed">Return Book</button>
          )}
          <button onClick={() => toast.success("Book renewed!")} disabled={!foundBook} className="px-6 py-2.5 bg-blue-600 text-white font-bold rounded-xl hover:bg-blue-700 transition-all disabled:opacity-40 disabled:cursor-not-allowed">Renew Book</button>
        </div>
      </div>

      {/* History */}
      <div className="bg-card rounded-2xl border border-border shadow-sm overflow-x-auto">
        <div className="px-5 py-4 border-b border-border">
          <h2 className="font-bold text-foreground" style={{ fontFamily: "Poppins, sans-serif" }}>Recent Transactions</h2>
        </div>
        <table className="w-full text-sm min-w-[560px]">
          <thead>
            <tr className="bg-muted text-xs text-muted-foreground uppercase border-b border-border">
              {["Book", "Member", "Issued", "Due", "Status", "Fine"].map((h) => (
                <th key={h} scope="col" className="px-5 py-3 text-left font-medium">{h}</th>
              ))}
            </tr>
          </thead>
          <tbody>
            {TRANSACTIONS.map((t) => (
              <tr key={t.id} className="border-b border-border/50 hover:bg-muted/30 transition-colors">
                <td className="px-5 py-3 font-semibold text-foreground max-w-[150px] truncate">{t.bookTitle}</td>
                <td className="px-5 py-3 text-foreground">{t.memberName}</td>
                <td className="px-5 py-3 text-xs text-muted-foreground" style={{ fontFamily: "JetBrains Mono, monospace" }}>{fmt(t.issueDate)}</td>
                <td className="px-5 py-3 text-xs text-muted-foreground" style={{ fontFamily: "JetBrains Mono, monospace" }}>{fmt(t.dueDate)}</td>
                <td className="px-5 py-3"><StatusBadge status={t.status} /></td>
                <td className="px-5 py-3 font-bold">{t.fine > 0 ? <span className="text-red-500">₹{t.fine}</span> : <span className="text-muted-foreground/30">—</span>}</td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  );
}

// ─── Screen: Reservations ─────────────────────────────────────────────────────

function ReservationsScreen({ role }: { role: UserRole }) {
  const [resv, setResv] = useState(RESERVATIONS_INIT);
  const [tab, setTab] = useState<"pending" | "approved" | "expired">("pending");
  const filtered = resv.filter((r) => r.status === tab);
  const counts = { pending: resv.filter((r) => r.status === "pending").length, approved: resv.filter((r) => r.status === "approved").length, expired: resv.filter((r) => r.status === "expired").length };

  const approve = (id: string) => { setResv((p) => p.map((r) => r.id === id ? { ...r, status: "approved" } : r)); toast.success("Reservation approved."); };
  const reject = (id: string) => { setResv((p) => p.filter((r) => r.id !== id)); toast.error("Reservation rejected."); };

  return (
    <div className="space-y-5">
      <div className="flex gap-1 p-1 bg-card rounded-xl border border-border shadow-sm overflow-x-auto" role="tablist">
        {(["pending", "approved", "expired"] as const).map((t) => (
          <button key={t} role="tab" aria-selected={tab === t} onClick={() => setTab(t)} className={`px-4 py-2 text-sm font-semibold rounded-lg capitalize transition-all flex items-center gap-1.5 ${tab === t ? "bg-primary text-primary-foreground shadow-sm" : "text-muted-foreground hover:text-foreground"}`}>
            {t}
            {counts[t] > 0 && <span className={`text-xs px-1.5 py-0.5 rounded-full font-bold ${tab === t ? "bg-white/20" : "bg-muted"}`}>{counts[t]}</span>}
          </button>
        ))}
      </div>

      <div className="bg-card rounded-2xl border border-border shadow-sm overflow-x-auto">
        <table className="w-full text-sm min-w-[560px]">
          <thead>
            <tr className="bg-muted text-xs text-muted-foreground uppercase border-b border-border">
              <th scope="col" className="px-5 py-3.5 text-left font-medium">Member</th>
              <th scope="col" className="px-5 py-3.5 text-left font-medium">Book</th>
              <th scope="col" className="px-5 py-3.5 text-left font-medium">Requested</th>
              <th scope="col" className="px-5 py-3.5 text-left font-medium">Expiry</th>
              <th scope="col" className="px-5 py-3.5 text-left font-medium">Status</th>
              {role === "librarian" && <th scope="col" className="px-5 py-3.5 text-left font-medium">Actions</th>}
            </tr>
          </thead>
          <tbody>
            {filtered.map((r) => (
              <tr key={r.id} className="border-b border-border/50 hover:bg-muted/30 transition-colors">
                <td className="px-5 py-4 font-semibold text-foreground">{r.memberName}</td>
                <td className="px-5 py-4 text-foreground">{r.bookTitle}</td>
                <td className="px-5 py-4 text-xs text-muted-foreground" style={{ fontFamily: "JetBrains Mono, monospace" }}>{fmt(r.requestDate)}</td>
                <td className="px-5 py-4 text-xs text-muted-foreground" style={{ fontFamily: "JetBrains Mono, monospace" }}>{fmt(r.expiryDate)}</td>
                <td className="px-5 py-4"><StatusBadge status={r.status} /></td>
                {role === "librarian" && (
                  <td className="px-5 py-4">
                    <div className="flex gap-1.5 flex-wrap">
                      {r.status === "pending" && (
                        <>
                          <button onClick={() => approve(r.id)} className="px-2.5 py-1 text-xs font-bold bg-green-50 dark:bg-green-900/20 text-green-700 dark:text-green-400 rounded-lg hover:bg-green-100 transition-colors" aria-label={`Approve ${r.memberName}`}>Approve</button>
                          <button onClick={() => reject(r.id)} className="px-2.5 py-1 text-xs font-bold bg-red-50 dark:bg-red-900/20 text-red-600 dark:text-red-400 rounded-lg hover:bg-red-100 transition-colors" aria-label={`Reject ${r.memberName}`}>Reject</button>
                        </>
                      )}
                      <button onClick={() => toast.success(`Notification sent to ${r.memberName}.`)} className="px-2.5 py-1 text-xs font-bold bg-muted text-muted-foreground rounded-lg hover:bg-muted/70 transition-colors flex items-center gap-1" aria-label={`Notify ${r.memberName}`}>
                        <Send className="w-3 h-3" aria-hidden="true" /> Notify
                      </button>
                    </div>
                  </td>
                )}
              </tr>
            ))}
          </tbody>
        </table>
        {filtered.length === 0 && <EmptyState icon={Calendar} title={`No ${tab} reservations`} desc="All clear here!" />}
      </div>
    </div>
  );
}

                  </td>
                </tr>
              );

          </tbody>
        </table>
      </div>
    </div>
  );
}

// ─── Screen: Notifications ────────────────────────────────────────────────────

function NotificationsScreen() {
  const [notifs, setNotifs] = useState(NOTIFS_INIT);
  const unread = notifs.filter((n) => !n.read).length;
  const markRead = (id: string) => setNotifs((p) => p.map((n) => n.id === id ? { ...n, read: true } : n));
  const markAll = () => setNotifs((p) => p.map((n) => ({ ...n, read: true })));
  const dismiss = (id: string) => setNotifs((p) => p.filter((n) => n.id !== id));

  const iconCfg: Record<string, { icon: React.ComponentType<{ className?: string }>; cls: string }> = {
    due: { icon: Clock, cls: "bg-amber-50 dark:bg-amber-900/20 text-amber-600 dark:text-amber-400" },
    approved: { icon: CheckCircle, cls: "bg-green-50 dark:bg-green-900/20 text-green-600 dark:text-green-400" },
    fine: { icon: DollarSign, cls: "bg-red-50 dark:bg-red-900/20 text-red-500" },
    new: { icon: BookOpen, cls: "bg-blue-50 dark:bg-blue-900/20 text-blue-600 dark:text-blue-400" },
    system: { icon: Activity, cls: "bg-muted text-muted-foreground" },
  };

  return (
    <div className="space-y-4 max-w-2xl" aria-live="polite">
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <h2 className="text-base font-bold text-foreground" style={{ fontFamily: "Poppins, sans-serif" }}>Notifications</h2>
          {unread > 0 && <span className="px-2 py-0.5 bg-red-100 dark:bg-red-900/30 text-red-600 dark:text-red-400 text-xs font-bold rounded-full">{unread} new</span>}
        </div>
        {unread > 0 && <button onClick={markAll} className="text-sm text-primary font-semibold hover:underline">Mark all read</button>}
      </div>

      {notifs.length === 0
        ? <EmptyState icon={Bell} title="All caught up!" desc="No notifications at this time." />
        : (
          <div className="space-y-3">
            {notifs.map((n) => {
              const { icon: Icon, cls } = iconCfg[n.type] ?? iconCfg.system;
              return (
                <article key={n.id} className={`bg-card rounded-2xl p-4 border shadow-sm transition-colors ${!n.read ? "border-primary/30 bg-primary/5 dark:bg-primary/10" : "border-border"}`}>
                  <div className="flex gap-3">
                    <div className={`w-9 h-9 rounded-xl flex items-center justify-center flex-shrink-0 ${cls}`} aria-hidden="true">
                      <Icon className="w-4 h-4" />
                    </div>
                    <div className="flex-1 min-w-0">
                      <div className="flex items-start justify-between gap-2">
                        <p className={`text-sm font-bold ${!n.read ? "text-foreground" : "text-muted-foreground"}`}>
                          {n.title}
                          {!n.read && <span className="ml-2 inline-block w-2 h-2 bg-primary rounded-full align-middle" aria-label="Unread" />}
                        </p>
                        <div className="flex items-center gap-1.5 flex-shrink-0">
                          <span className="text-xs text-muted-foreground whitespace-nowrap">{n.time}</span>
                          <button onClick={() => dismiss(n.id)} className="text-muted-foreground/50 hover:text-muted-foreground transition-colors" aria-label={`Dismiss: ${n.title}`}>
                            <X className="w-3.5 h-3.5" />
                          </button>
                        </div>
                      </div>
                      <p className="text-xs text-muted-foreground mt-1 leading-relaxed">{n.message}</p>
                      {!n.read && <button onClick={() => markRead(n.id)} className="text-xs text-primary font-semibold mt-2 hover:underline">Mark as read</button>}
                    </div>
                  </div>
                </article>
              );
            })}
          </div>
        )}
    </div>
  );
}

// ─── Root App ─────────────────────────────────────────────────────────────────

export default function App() {
  const [screen, setScreen] = useState<Screen>("landing");
  const [role, setRole] = useState<UserRole>("student");
  const [selectedBook, setSelectedBook] = useState<Book | null>(null);
  const [darkMode, setDarkMode] = useState(false);
  const unread = NOTIFS_INIT.filter((n) => !n.read).length;

  const login = (r: UserRole) => { setRole(r); setScreen("dashboard"); };
  const logout = () => { setScreen("landing"); setSelectedBook(null); };
  const nav = (s: Screen) => { if (s !== "book-details") setSelectedBook(null); setScreen(s); };
  const viewBook = (b: Book) => { setSelectedBook(b); setScreen("book-details"); };
  const toggleDark = () => setDarkMode((d) => !d);

  const titles: Record<Screen, string> = {
    landing: "", login: "", register: "", forgot: "",
    dashboard: role === "student" ? "Student Dashboard" : "Admin Dashboard",
    browse: "Browse Books",
    "book-details": selectedBook?.title ?? "Book Details",
    "my-account": "My Account",
    "book-management": "Book Management",
    "member-management": "Member Management",
    "issue-return": "Issue & Return",
    reservations: "Reservations",
    fines: "Fine Management",
    reports: "Reports & Analytics",
    settings: "Settings",
    notifications: "Notifications",
    profile: "My Profile",
  };

  const toasterPos = "top-right" as const;

  if (screen === "landing") return (
    <div className={darkMode ? "dark" : ""}>
      <Toaster richColors position={toasterPos} />
      <LandingPage onGetStarted={() => setScreen("register")} onLogin={() => setScreen("login")} darkMode={darkMode} onToggleDark={toggleDark} />
    </div>
  );
  if (screen === "login") return (
    <div className={darkMode ? "dark" : ""}>
      <Toaster richColors position={toasterPos} />
      <LoginScreen onLogin={login} onRegister={() => setScreen("register")} onForgot={() => setScreen("forgot")} />
    </div>
  );
  if (screen === "register") return (
    <div className={darkMode ? "dark" : ""}>
      <Toaster richColors position={toasterPos} />
      <RegisterScreen onSuccess={() => setScreen("login")} onLogin={() => setScreen("login")} />
    </div>
  );
  if (screen === "forgot") return (
    <div className={darkMode ? "dark" : ""}>
      <Toaster richColors position={toasterPos} />
      <ForgotPassword onBack={() => setScreen("login")} />
    </div>
  );

  const renderContent = () => {
    switch (screen) {
      case "dashboard": return role === "student" ? <StudentDashboard onNavigate={nav} /> : <LibrarianDashboard onNavigate={nav} />;
      case "browse": return <BrowseBooks onViewBook={viewBook} />;
      case "book-details": return selectedBook ? <BookDetails book={selectedBook} onBack={() => nav("browse")} /> : null;
      case "my-account": return <MyAccount />;
      case "book-management": return <BookManagement />;
      case "member-management": return <MemberManagement />;
      case "issue-return": return <IssueReturnScreen />;
      case "reservations": return <ReservationsScreen role={role} />;
      case "fines": return <FineManagement role={role} />;
      case "reports": return <ReportsAnalytics />;
      case "settings": return <SettingsPage />;
      case "notifications": return <NotificationsScreen />;
      case "profile": return <ProfileScreen role={role} onToggleDark={toggleDark} darkMode={darkMode} />;
      default: return null;
    }
  };

  return (
    <div className={darkMode ? "dark" : ""}>
      <Toaster richColors position={toasterPos} />
      <AppShell role={role} screen={screen} onNavigate={nav} onLogout={logout} title={titles[screen]} unread={unread} darkMode={darkMode} onToggleDark={toggleDark}>
        {renderContent()}
      </AppShell>
    </div>
  );
}
